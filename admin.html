// 1. นำเข้าโมดูล Firebase ที่จำเป็น import { initializeApp } from "https://www.gstatic.com/firebasejs/10.11.0/firebase-app.js"; import { getFirestore, doc, setDoc } from "https://www.gstatic.com/firebasejs/10.11.0/firebase-firestore.js"; import { getStorage, ref, uploadBytes, getDownloadURL } from "https://www.gstatic.com/firebasejs/10.11.0/firebase-storage.js"; // 2. การตั้งค่า Firebase (กูเร่เช็ค API Key ของตัวเองอีกทีนะ) const firebaseConfig = { apiKey: "AIzaSy...", // เอา API Key จากโปรเจกต์ AquaSmart-V2 มาใส่ตรงนี้ authDomain: "aquasmart-v2.firebaseapp.com", projectId: "aquasmart-v2", storageBucket: "aquasmart-v2.firebasestorage.app", // ต้องลงท้ายด้วย .app ตามที่เซ็ตไว้ messagingSenderId: "482478104079", appId: "1:482478104079:web:..." }; // เริ่มต้นการทำงาน const app = initializeApp(firebaseConfig); const db = getFirestore(app); const storage = getStorage(app); // 3. ดักจับเหตุการณ์ตอนกดปุ่มบันทึก document.getElementById('btnSave').addEventListener('click', async () => { const nameTH = document.getElementById('fishNameTH').value; const nameEN = document.getElementById('fishNameEN').value; const description = document.getElementById('fishDescription').value; const imageFile = document.getElementById('fishImageFile').files[0]; // เช็คข้อมูลเบื้องต้น if (!nameTH || !nameEN || !imageFile) { alert("กูเร่! กรอกชื่อปลาและเลือกรูปให้ครบก่อนนะ"); return; } console.log("เริ่มกระบวนการบันทึก..."); const btn = document.getElementById('btnSave'); btn.disabled = true; btn.innerText = "⏳ กำลังบันทึก..."; try { // A. อัปโหลดรูปภาพไปยัง Firebase Storage const storageRef = ref(storage, 'fish_samples/' + nameEN + '.jpg'); const uploadTask = await uploadBytes(storageRef, imageFile); const imageURL = await getDownloadURL(uploadTask.ref); console.log("อัปโหลดรูปสำเร็จ: " + imageURL); // B. บันทึกข้อมูลข้อความลง Firestore await setDoc(doc(db, "fish_info", nameEN), { thai_name: nameTH, english_name: nameEN, description: description, image_url: imageURL, updated_at: new Date() }); console.log("บันทึกข้อมูลลงฐานข้อมูลสำเร็จ!"); alert("✅ บันทึกข้อมูลปลา " + nameTH + " เรียบร้อยแล้ว!"); // ล้างฟอร์มหลังจากบันทึกเสร็จ location.reload(); } catch (error) { console.error("เกิดข้อผิดพลาด:", error); alert("❌ บันทึกไม่สำเร็จ: " + error.message); } finally { btn.disabled = false; btn.innerText = "💾 บันทึกข้อมูลลงฐานข้อมูล"; } });
